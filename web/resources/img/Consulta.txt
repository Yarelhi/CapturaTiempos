SELECT u.Status AS Stat, v.VALUE_LIST_DETAIL_VALUE, v.VALUE_LIST_DETAIL_ID AS VID
FROM xxeks_users_all AS u, xxeks_values_lists_details AS v
WHERE VALUE_LIST_DETAIL_VALUE = ( 
SELECT v.VALUE_LIST_DETAIL_VALUE
FROM xxeks_values_lists_details AS v
WHERE v.VALUE_LIST_DETAIL_ID = u.Status )
